#Reload Public Unrest - should always incur a 5% penalty
PLA_reload_public_unrest = {

	clamp_variable = {
		var = PLA_public_unrest_percent
		min = 0
		max = 100
	}
	set_variable = {
		var = PLA_public_unrest_effect
		value = PLA_public_unrest_percent
	}
	multiply_variable = {
		var = PLA_public_unrest_effect
		value = -0.004
	}
	add_to_variable = {
		var = PLA_public_unrest_effect
		value = -0.05
	}
	
}
