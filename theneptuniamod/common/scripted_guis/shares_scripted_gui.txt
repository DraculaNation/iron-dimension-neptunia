scripted_gui = {
  shares_gui = {
    window_name = "shares_gui_window"
    context_type = player_context
	
    visible = {
		has_country_flag = view_shares

    }

    effects = {
		close_button_click = { clr_country_flag = view_shares }
    }
	
	properties = {
		planeptune_progress_bar = { frame = var:PLA.shares_factor }
		lastation_progress_bar = { frame = var:LAS.shares_factor }
		leanbox_progress_bar = { frame = var:LEA.shares_factor }
		lowee_progress_bar = { frame = var:LOW.shares_factor }
	}
	
  }
  open_button_gui = {
	window_name = "open_button_window"
	context_type = player_context
	parent_window_token = top_bar
	
	visible = {
		always = yes
	}
	
	effects = {
		open_button_click = {
			if = {
				limit = { has_country_flag = view_shares }
				clr_country_flag = view_shares
			}
			else = { set_country_flag = view_shares }
		}
	}
  }
